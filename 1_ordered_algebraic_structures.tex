          F^{-1}(F(x) \ \IN^{\prime} \ F(y)) &=& F^{-1}(F(x \ \IN \ y)) \\
          &=& x \ \IN \ y \\
          &=& F^{-1}(F(x)) \ \IN \ F^{-1}(F(y))
        \end{eqnarray*}

        \begin{eqnarray*}
          F^{-1}(0)
        \end{eqnarray*}

        \begin{eqnarray*}
          F^{-1}(1)
        \end{eqnarray*}
      \end{multicols}

      \PN Luego, $F^{-1}$ es homomorfismo y por lo tanto $F$ es isomorfismo.
  \end{proof}
